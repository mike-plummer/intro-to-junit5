public class whenRetrievingByStartingCharacter {
    
    @Nested
    public class withMatchingValue {
        @Test
        public void shouldReturnMatchingValues() {
            assertEquals(dataBean.getStatesStartingWith('A').size(), 4);
        }

        @Test
        public void shouldReturnSortedValues() {
            List<String> sortedMatches = 
                new ArrayList<>(dataBean.getStatesStartingWith('A'));
            sortedMatches.sort();
            assertEquals(sortedMatches, dataBean.getStatesStartingWith('A'));
        }
    }

    @Nested
    public class withNonMatchingValue {
        @Test
        public void shouldReturnEmptyList() {
            assertTrue(dataBean.getStatesStartingWith('!').isEmpty());
        }
    }
}
